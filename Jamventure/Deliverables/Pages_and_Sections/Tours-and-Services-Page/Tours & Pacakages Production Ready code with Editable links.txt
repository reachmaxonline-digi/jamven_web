<script>
    function filterTours(category, btnElement) {
        // 1. Manage Button Styles
        const buttons = document.querySelectorAll('#jv-tours-wrapper .filter-btn');
        buttons.forEach(btn => {
            btn.classList.remove('tab-active');
            // Reset to default style
            btn.classList.add('tab-inactive');
        });

        // Add 'tab-active' class to clicked button
        btnElement.classList.add('tab-active');
        btnElement.classList.remove('tab-inactive');

        // 2. Filter the Grid
        const cards = document.querySelectorAll('#jv-tours-wrapper .tour-card');
        const noResults = document.getElementById('no-results');
        let visibleCount = 0;

        cards.forEach(card => {
            // Reset animation
            card.classList.remove('animate-fade');
            void card.offsetWidth; // trigger reflow

            const match = category === 'all' || card.getAttribute('data-category') === category;

            if (match) {
                card.style.display = 'flex';
                card.classList.add('animate-fade');
                visibleCount++;
            } else {
                card.style.display = 'none';
            }
        });

        // 3. Show/Hide "No Results" message
        if (visibleCount === 0) {
            noResults.classList.remove('hidden');
            document.getElementById('tours-grid').parentElement.classList.add('hidden');
        } else {
            noResults.classList.add('hidden');
            document.getElementById('tours-grid').parentElement.classList.remove('hidden');
        }
    }
</script>
